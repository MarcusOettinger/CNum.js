 12 /**
 13  * @class CNum.js
 14  * is a javascript object plotting a representation of a number z in the complex
 15  * plane. The idea is to set a complex value via an algebraic expression, 
 16  * transform it into different bases and draw a Gaussian plot (aka Argand
 17  * diagram).
 18  * Here's a <a href="https://MarcusOettinger.github.io/CNum.js/">demo page</a>.<br>
 19  * License: <a href="http://opensource.org/licenses/MIT">The MIT License</a>.<br>
 20  * <br><br>
 21  * Graphics are drawn onto a html5 canvas - this should nowadays
 22  * be supported  by most browsers.<br><br>
 23  * I wrote the code because I needed a dynamic plot of complex values in
 24  * webpages for a a basic maths lecture. It serves a purpose and is far from
 25  * being cleanly written, nicely formatted or similar.
 26  * 
 27  * <br><br>
 28  * CNum.js uses some external libraries:
 29  * <ul>
 30  * <li>jquery: [jquery]{http://jquery.org}</li>
 31  * <li>jcanvas: 
 32  * {@link http://calebevans.me/projects/jcanvas/ jcanvas}
 33  * (drawing routines)</li>
 34  * <li>mathjs: {@link http://mathjs.org mathjs} (that can do much more!)</li>
 35  * </ul>
 36  * Creates a CNum object - the complex value is set via an algebraic expression.
 37  * <br><br>
 38  * Usage is simple: object = new CNum( expression ) creates a new complex
 39  * number,<br>
 40  * e.g. <b><i> z = new CNum("2+i"); </i></b>   
 41  * will create <b><i>z</i></b> with the value <b><i>2+i</i></b><br>.
 42  * @param {string} expr a text representing a complex number. Cartesian, trigonometric and polar expressions can be used, e.g. '2+3i' or '12*exp(3i)'
 43  * <br><br>
 44 */
 45 function CNum(expr) {
 46         // Private:
 47         // -----------------------------------------------------------------------------
 48         // paranoia mode: some reasonable defaults
 49         var _margin = 50;	// margin around the plot
 50         var X_C = 100;		// center coordinates
 51         var Y_C = 100;		// (will be overwritten by setScale)
 52         var _plotW;
 53         var _plotH;
 54         var _SCALE = 10.0;	//
 55         var _maxradius = 1;	// (value the plot was autoscaled to)
 56         var _tickLength= 3;	//
 57         var _daCanvas = null;	// the canvas to use
 58         var _count = 0;		// count the number of arrows on cnv - needed to display
 59 			// angles in a readable form
 60         var _TICKSCALE= 0.001;
 61 
 62         // default options - values are settable for every number displayed
 63 	/**
 64 	 * Options used to change the appearance of a plot. This set of options
 65 	 * is used in all methods displaying numbers.
 66 	 * @name options
 67 	 * @memberOf CNum
 68 	 * @function
 69          * @param {Object} options
 70          * @param {boolean} options.clear - remove canvas contents before drawing
 71          * @param {boolean} options.coords - draw a coordinate system
 72 	 * @param {boolean} options.Arrow draw <i><b>z</b></i> as an arrow - if set to false, draw a point. Defaults to true.
 73 	 * @param {boolean} options.circle draw a circle with radius <i><b>|z|</b></i>. Defaults to false.
 74 	 * @param {boolean} options.ReIm show real and imaginary part in the plot. Default is true.
 75 	 * @param {boolean} options.Ctext print <i><b>z</b></i> in cartesian notation. Defaults to true
 76 	 * @param {boolean} options.arc show angle and value as an arc. Default is true.
 77 	 * @param {string} options.color set a color to use for arrow, text, angle. Defaults to '#888'.
 78 	 * @param {boolean} options.colarrow draw arrow and text in the selected color (default is black).
 79 
 80 	 */
 81 
 82         var _defaults = { 
 83             		// general plot options
 84             		clear: false, 		// clear canvas
 85             		coords: false, 		// draw coordinate system
 86 
 87             		// options for complex numbers
 88             		Arrow: true, 		// draw an arrow - if set to false, draw a point
 89             		circle: false, 		// draw a circle with radius |z|
 90             		ReIm: true, 		// show real and imaginary part
 91             		Ctext: true,		// print z in cartesian notation
 92             		arc: true, 		// show angle and value as an arc
 93 
 94             		// colors
 95             		color: "#888", 		// color to use for arrow, text, angle
 96             		colarrow: false 		// colored arrow and text (default is black)
 97         };
 98 
 99 
100         // math magic miracles...
101         zStr = math.eval( expr );			// eval expression expr to create an
102 					// expression mathjs can handle for sure...
103         this.zNum = math.complex(zStr);		// ... and create a complex number
104         this.x = math.round(this.zNum.re, 3);	// store Re(z) and Im(z) for later
105         this.y = math.round(this.zNum.im, 3);	//
106         var _radius = this.zNum.toPolar().r;		// find absolute value (radius)
107         var _a = this.zNum.toPolar().phi;		// get angle phi - bummer:
108         this.angle = _a<0?2*Math.PI+_a:_a; 		// mathjs treats angles > PI as negative
109 						//  - great for calculations but
110 						// B*A*D if drawing polar plots...  
111 
112         // calculate cartesian coordinates of a given point (x/y) on the canvas
113         function _XScaled(x) { return _SCALE * x; };
114         function _YScaled(y) { return _SCALE * y; };
115         function _XPos(x) { return X_C + _XScaled(x); };
116         function _YPos(y) { return Y_C - _YScaled(y); };
117 
118         // return the sign of a number x (this ought to be fast and safe)
119         function sign(x) { return typeof x === 'number' ? x ? x < 0 ? -1 : 1 : x === x ? 0 : NaN : NaN; }
120         // return the largest of three values
121         function getmax(v1, v2, v3) { m = v1>v2?v1:v2; return v3>m?v3:m; };
122 
123         // return minimum in the sense of: b if b<a, a otherwise.
124         function min(a,b){ if (b<a){ return b; } return a; };
125 
126         // return maximum in the sense of: b if b>a, a otherwise.
127         function max(a,b){ if (b>a){ return b; } return a; };
128 
129         // SetScale: calculate a reasonable scale for the canvas to plot a complex number
130         // with given radius
131         // (used by displayGauss() and displayGaussWithRadius()).
132         function setScale(maxradius) {
133                     X_C = _daCanvas.width() / 2;
134                     Y_C = _daCanvas.height() / 2;
135                     _plotW = _daCanvas.width() - 2 * _margin;
136                     _plotH = _daCanvas.height() - 2 * _margin;
137                     _SCALE = min(_plotW, _plotH) / (2 * maxradius);
138                     _maxradius = maxradius;
139 
140                     while (maxradius/_TICKSCALE > 10) {
141                        _TICKSCALE= _TICKSCALE*10; 
142                     }
143                     if (_TICKSCALE >10) _TICKSCALE= _TICKSCALE*2;
144         } // SetScale
145 
146 
147         // drawticks(pos): draw ticks on x-and y-axis at position x = pos / y = pos
148         // (internal function used by axis())
149         //
150         function drawticks(pos) {
151                 _daCanvas.drawLine({ strokeStyle: '#000', strokeWidth: 2, rounded: false,
152                         endArrow: false, x1: _XPos(pos) , y1: Y_C, x2: _XPos(pos), y2: Y_C + _tickLength });
153                 _daCanvas.drawText({ strokeStyle: '#000', strokeWidth: 1,
154                         x: _XPos(pos), y: Y_C+10, fontSize: 10, fontFamily: 'serif',
155                         text: pos });
156                 _daCanvas.drawLine({ strokeStyle: '#000', strokeWidth: 2, rounded: false,
157                         endArrow: false, x1: X_C , y1: _YPos(pos), x2: X_C - _tickLength, y2:                                                         _YPos(pos) });
158                 _daCanvas.drawText({ strokeStyle: '#000', strokeWidth: 1,
159                         x: X_C-20, y: _YPos(pos)+3, fontSize: 10, fontFamily: 'serif',
160                         text: pos });
161         }; // drawticks
162 
163         // draw axes for a cartesian coordinate system
164         //
165         function axis(){
166                if (_daCanvas == null) return;
167 
168                _daCanvas.drawLine({ strokeStyle: '#000', strokeWidth: 2, rounded: false,
169                        endArrow: true, arrowRadius: 10, arrowAngle: 90, 
170                        x1: _margin, y1: Y_C, x2: _plotW + _margin , y2: Y_C });
171                _daCanvas.drawLine({ strokeStyle: '#000', strokeWidth: 2, rounded: false,
172                        endArrow: true, arrowRadius: 10, arrowAngle: 90,
173                        x1: X_C, y1: _plotH+_margin,  x2: X_C, y2: _margin });
174                 for ( i=_TICKSCALE; i<=_maxradius; i+=_TICKSCALE) {
175                     drawticks(i);
176                 }
177         }; // axis
178 
179 
180         // draw x/y, real and imaginary part using thin lines and style
181         // 
182         function drawReIm(x, y, style) {
183                 xpos = _XPos(x);
184                 ypos = _YPos(y);
185                 // Draw arrow from center to P(xpos, ypos)
186                 _daCanvas.drawLine({ strokeStyle: style, strokeWidth: 1,
187                         x1: xpos, y1: Y_C, x2: xpos, y2: ypos });
188                 _daCanvas.drawLine({ strokeStyle: style, strokeWidth: 1,
189                         x1: X_C, y1: ypos, x2: xpos, y2: ypos });
190         }; // drawReIm
191 
192 
193 
194         // draw a complex number z= x + iy in gaussian / argand plane.
195         // (polar representation of coordinates x/y in cartesian system.
196         // x,y is the starting point of the free vector to draw.
197         // as arrow/vector)
198         //
199         function drawz(z, settings, xstart, ystart) {
200                 // Draw arrow from A(xstart, ystart) to B(xpos, ypos)
201                 xpos = _XPos(z.x);
202                 ypos = _YPos(z.y);
203                 x = _XScaled(xstart);
204                 y = _YScaled(ystart);
205 
206                 style = settings.colarrow?settings.color:"#000";
207                 if (settings.Arrow)
208                     _daCanvas.drawLine({ strokeStyle: style, strokeWidth: 2, rounded: false,
209                         endArrow: true, arrowRadius: 10, arrowAngle: 45, 
210                         x1: X_C + x, y1: Y_C - y, x2: xpos + x, y2: ypos - y });
211                 else
212                     _daCanvas.drawRect({ strokeStyle: style, strokeWidth: 2, fromCenter: true,
213                        x: xpos, y: ypos, width: 5, height: 5 });
214 
215                 if (settings.Ctext) 
216                     _daCanvas.drawText({ strokeStyle: style, strokeWidth: 1,
217                             x: xpos + x +10, y: ypos-(sign(z.y)*10) - y, fontSize: 12, fontFamily: 'serif',
218                             text: 'z =' + math.complex(z.x, z.y).toString() });
219         }; //drawz
220 
221 
222 
223         // drawAngle: draw an angle of the arrow (mathematically starting from x-axis)
224         function drawAngle(angle, settings) {
225 
226                 style = settings.colarrow?settings.color:"#000";
227 
228                 _daCanvas.drawArc( { strokeStyle: style, x: X_C, y: Y_C,
229                         radius: 50+ _count*2, start: (Math.PI/2-angle), end: Math.PI/2, inDegrees: false });
230                 tx = X_C + 30*math.cos(angle - Math.PI/6);
231                 ty = Y_C - 30*math.sin(angle - Math.PI/6);
232                 _daCanvas.drawText({ strokeStyle: style, strokeWidth: 1,
233                         x: tx,  y: ty, fontSize: 12, fontFamily: 'serif', text: math.round(angle,2) });
234         }; // drawAngle
235 
236 
237         // draw a circle with radius |z| 
238 	// (to clarify the picture of rotating pointers in the complex plane)
239         function circ() {
240                 if (_daCanvas == null) return;
241                 _daCanvas.drawArc({ draggable: false, strokeStyle: '#999', strokeWidth: 1, 
242                      x: X_C, y: Y_C, radius: (_radius * _SCALE)
243                 });
244         }; // circ
245

247         // =======================================================
248         // Public:
249         // =======================================================
250         // return radius of the complex cartesian x+iy in gaussian plane
251 	/** Return radius of the pointer
252 	 * @return {float} radius of the complex number (norm)
253 	 */
254         this.getradius = function() { return _radius;  };
255 
256 
257         // Return complex number z in cartesian coordinates as html string.
258         //
259         /** Return a string of the complex number z in cartesian representation (e.g. '1+2i')
260          *  @return {string} a string containing the simplest form of the complex number <b><i>z = x + i·y</i></b>
261          */
262         this.ToCartesian = function() {
263                 retval = this.zNum.format(3);
264                 return retval;
265         };
266 
267         // Return complex conjugate as html.
268         //
269         /** Return a string of the complex conjugate z* in cartesian representation (e.g. '1-2i')
270          *  @return {string} <b><i>z̅</i></b>, a  string containing the complex conjugate of <b><i>z</i></b>.
271          */
272         this.ToCC = function() {
273                 retval = math.conj(this.zNum).format(3);
274                 return retval;
275         };
276 
277  
278         // Return norm as html.
279         //
280         /** Return a string of the norm (length of the vector) of the complex number z (e.g. '2.236')
281          *  @return {string} a string containing the norm as a real number.
282          */
283         this.ToR = function() {
284                 retval = math.round(this.zNum.toPolar().r,3);
285                 return retval;
286         };
287 
288         // Return complex number z in trigonometric representation as html.
289         //
290         /** Return a string of the complex number z in trigonometric representation (e.g. '2.236(cos(1.107) + i · sin(1.107))')
291          *  @return {string} a string containing the complex number as a mathematical expression in trigonometric notation.
292          */
293         this.ToTrigonometric = function() {
294                 rad = math.round(this.zNum.toPolar().r,3);
295                 phi = math.round(this.zNum.toPolar().phi,3);
296                 retval = rad + "·(cos(" + phi + ") + i ·sin(" + phi +"))";
297                 return retval;
298         };
299 
300         // Return complex number z in polar coordinates as html.
301         //
302         /** Return a string of the complex number z in polar representation (e.g. '2.236 e<sup>i 1.107</sup>')
303          *  @return {string} a string containing the number <b><i>z</i></b> in polar coordinates with a bit of html for the superscripts.
304          */
305         this.ToPolar = function() {
306                 rad = math.round(this.zNum.toPolar().r,3);
307                 phi = math.round(this.zNum.toPolar().phi,3);
308                 retval = rad + "· e<sup>i " + phi + "</sup>";
309                 return retval;
310         };

312         // the good stuff: Graphics
313         // -----------------------------------------------------------------------------
314         /** Set the canvas to plot on
315          * @param {canvas object} cnv - the html canvas to draw on
316 	 * (conveniently selected via jquery).
317          */
318         this.setCanvas = function( cnv ) {  _daCanvas = cnv; };
319 
320         // get cartesian coordinates of a given point (x/y) on the canvas
321         this.getXPos = function(x) { return _XPos(x); };
322         this.getYPos = function(y) { return _YPos(y); };
323 
324 
325         // add product of zs to plot - add to existing plot
326 	/** multiply two complex numbers in a Gaussian plot.
327 	 * @param {CNum} z2: the number to multiply with
328 	 * @param {options} options set appearance of the complex number to add (see {@link CNum.options} for a list of possible settings)
329 	 * @param {options} resultoptions change appearance of the resulting complex number (see {@link CNum.options} for a list of possible settings)
330 	 * @return {CNum} result of the operation as a CNum
331 	 */
332         this.multiplyCNum =function(z2, options, resultoptions){
333             resultoptions = ( typeof resultoptions == 'undefined' ? options : resultoptions);
334 
335             // calculate resulting value z = this + z2
336             res = new CNum( math.multiply(this.zNum, z2.zNum).toString() );
337 
338             // replot this and rescale to largest absolute value
339             this.displayGauss(options, getmax(this.getradius(), z2.getradius(), res.getradius()));	
340             this.addToPlot(z2, options);		// multiplication of this and s2
341             this.addToPlot(res, resultoptions);		// display the result
342             return res;
343         }
344 
345 
346         // add sum of zs to plot - add to existing plot
347 	/** show the sum of two complex numbers in a Gaussian plot.
348 	 * @param {CNum} z2: the number to add
349 	 * @param {options} options set appearance of the complex number to add (see {@link CNum.options} for a list of possible settings)
350 	 * @param {options} resultoptions change appearance of the resulting complex number (see {@link CNum.options} for a list of possible settings)
351 	 * @return {CNum} result of the operation as a CNum
352 	 */
353         this.addCNum =function(_zToAdd, options, resultoptions){
354             resultoptions = ( typeof resultoptions == 'undefined' ? options : resultoptions);
355 
356             // calculate resulting value z = this + _zToAdd
357             res = new CNum( this.zNum.toString() + "+" + _zToAdd.zNum.toString() );
358             // replot this and rescale to largest absolute value
359             this.displayGauss(options, getmax(this.getradius(), _zToAdd.getradius(), res.getradius()));	
360 
361             this.addVector(_zToAdd, this.x, this.y, options);		// vector addition of this and _zToAdd
362             this.addToPlot(res, resultoptions);			// display the result
363             return res;
364         }

367         /** add a CNum to an existing plot - may be used to plot several
368 	 *numbers on one canvas at the same time.
369          * Caveat: the plot will not be rescaled nor cleared (of course).
370          * @param {CNum} z the complex number to add to the current plot
371 	 * @param {options} options change plot appearance (see {@link CNum.options} for a list of possible settings)
372         */
373         this.addToPlot = function( z, options ) {
374                 // handle defaults
375 //                var defaults = { clear: false, coords: false, circle: false, ReIm: true, color: "#888", colarrow: false, arc: true };
376                 var settings = $.extend( {}, _defaults, options );
377                 if (settings.arc) drawAngle(z.angle, settings);
378                 if (settings.ReIm) drawReIm(z.x, z.y, settings.color);
379 
380                 drawz( z, settings, 0, 0 );
381                 _count++;
382         }; // addToPlot
383 
384         /** add a CNum as a free vector to an existing plot - used to plot
385 	* several numbers on a single canvas. This method plots a CNum starting
386 	* at a point (x/y), for example to display sums of complex numbers.
387         * Caveat: the plot will not be rescaled nor cleared (of course).
388         * @param {CNum} z - the complex number to add
389         * @param {coordinate} x,y coordinates of the starting point
390 	* @param {options} 
391 	* change plot appearance (see {@link CNum.options} for a list of possible settings)
392         */
393         this.addVector = function( z, x, y, options ) {
394                 var settings = $.extend( {}, _defaults, options );
395                 if (settings.arc) drawAngle(z.angle, settings);
396                 if (settings.ReIm) drawReIm(z.x, z.y, settings.color);
397                drawz( z, settings, x , y );
398                 _count++;
399         }; // addToPlot
400

402 
403         // * clear canvas and draw 
404         // 
405         /** Draw the complex number z as an arrow in the complex plane on the
406 	 * selected canvas (set by setCanvas()).
407 	 * This method either plots a single CNum or the first one in a series
408 	 * of CNums drawn onto the same canvas. Additional numbers can be
409 	 * displayed on that canvas using addToPlot
410 	 * @param {options} options change plot appearance (see {@link CNum.options} for a list of possible settings)
411         * @param {optional number} radius scale the plot for a complex number of this value (default: autoscale)
412          *@see setCanvas(): Set the canvas to draw on.
413          *@see addToPlot(): draw another number into an existing plot.
414         */
415         this.displayGauss = function(options, radius){
416                 // handle defaults
417                 maxR = ( typeof radius == 'undefined' ? _radius : radius);
418                 var settings = $.extend( {}, _defaults, { clear:true, coords: true, circle: true }, options );
419 
420                 setScale(maxR);
421                 if (settings.clear) {
422                     _daCanvas.clearCanvas();
423                     _count = 0;
424                 }
425                 if (settings.coords) axis();
426                 if (settings.circle) circ();
427 
428                 this.addToPlot(this, settings); 
429         };
430 
431         // return a brandnew CNum :-)
432         return this;
